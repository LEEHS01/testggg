ALTER PROCEDURE [dbo].[GET_SENSOR_STEP]
(
    @obsid int
)
AS
BEGIN
    SELECT 
        @obsid as obsid,
        ISNULL(MEAS_ADDINFO, '25') as toxistep,
        ISNULL(MEAS_ADDINFO, '25') as chemistep
    FROM (
        SELECT TOP 1 MEAS_ADDINFO
        FROM TB_3D_DATA
        WHERE OBSIDX = @obsid
        ORDER BY OBSDT DESC
    ) as latest
END
